以下是部份对数函數的[積分表](../Page/積分表.md "wikilink")(書寫時省略了不定積分結果中都含有的任意常數*C<sub>n</sub>*)。

\(\begin{cases}
\log_\alpha^2x=\log_\alpha\log_\alpha x\\
\log_\alpha^2|x|=\log_\alpha|(\log_\alpha|x|)|
\end{cases}\)

\[\begin{cases}
\int\ln\!x\ \mbox{d}x&=x\ln\!x-\int x\!\dfrac{1}{x}\ \mbox{d}x\\
&=x\ln\!x-\int\!\!\mbox{d}x\\
&=x\ln\!x-x\\
\int\ln\!-x\ \mbox{d}x&=-\int x\ln\!-x\ \mbox{d}\!-\!x\\
&=-\left(\int-x\ln\!-x+x\right)\\
&=x\ln\!-x-x
\end{cases}
\therefore
\begin{cases}
\int\ln\!|x|&=x\ln\!|x|-x\\
\int\log_\alpha\!|x|\mbox{d}x&=\int\dfrac{\ln\!|x|}{\ln\!\alpha}\mbox{d}x\\
&=\dfrac{x\ln\!|x|-x}{\ln\!\alpha}\\
&=x\log_\alpha\!|x|-\dfrac{x}{\ln\!\alpha}
\end{cases}\]

\[\int\ln\!Mx\mbox{d}x=x\ln\!Mx-x\]

\[\int(\ln\!x)^2\mbox{d}x=x(\ln\!x)^2-2x\ln\!x+2x\]

\[\int(\ln\!Mx)^n\mbox{d}x=x(\ln\!Mx)^n-n\int(\ln\!Mx)^{n-1} dx\]

\[\int\frac{\mbox{d}x}{\ln\!x}=\ln\!|\!\ln\!x|+\ln\!x+\sum^\infty_{I=2}\frac{(\ln\!x)^I}{I!I}\]

\[\int\frac{\mbox{d}x}{(\ln\!x)^n}=-\frac{x}{(n-1)(\ln\!x)^{n-1}}+\frac{1}{n-1}\int\frac{\mbox{d}x}{(\ln x)^{n-1}}\qquad(n\neq1)\]

\[\int x^N\ln\!x\mbox{d}x=x^{N+1}\left(\frac{\ln\!x}{N+1}-\frac{1}{(N+1)^2}\right)\qquad(N\neq-1)\]

\[\int x^N(\ln\!x)^n\ \mbox{d}x=\frac{x^{N+1}(\ln\!x)^n}{N+1}-\frac{n}{N+1}\int x^N(\ln x)^{n-1}dx\qquad(N\neq-1)\]

\[\int\frac{(\ln\!x)^n\ \mbox{d}x}{x}=\frac{(\ln\!x)^{n+1}}{n+1}\qquad(n\neq-1)\]

\[\int\frac{\ln\!x\ \mbox{d}x}{x^r}=-\frac{\ln\!x}{(r-1)x^{r-1}}-\frac{1}{(1-r)^2x^{r-1}}\qquad(r\neq1)\]

\[\int\frac{(\ln\!x)^n\ \mbox{d}x}{x^r}=-\frac{(\ln x)^n}{(r-1)x^{r-1}}+\frac{n}{r-1}\int\frac{(\ln x)^{n-1}dx}{x^m}\qquad(m\neq1)\]

\[\int\frac{x^N\ \mbox{d}x}{(\ln\!x)^n}=-\frac{x^{N+1}}{(n-1)(\ln\!x)^{n-1}}+\frac{N+1}{n-1}\int\frac{x^N\mbox{d}x}{(\ln\!x)^{n-1}}\qquad(n\neq1)\]

\[\int\frac{\mbox{d}x}{x\ln\!x}=\ln\!|\!\ln\!x|\]

\[\int\frac{\mbox{d}x}{x^n\ln\!x}=\ln\!|\!\ln\!x|+\sum^\infty_{I=1}(-1)^I\frac{(n-1)^I(\ln\!x)^I}{I!I}\]

\[\int\frac{\mbox{d}x}{(x\ln\!x)^n}=\frac{1}{(1-n)(\ln\!x)^{n-1}} \qquad(n\neq1)\]

\[\int\sin\ln\!x\ \mbox{d}x=\frac{x(\sin\ln x-\cos\ln\!x)}{2}\]

\[\int\cos\ln\!x\ \mbox{d}x=\frac{x(\sin\ln x+\cos\ln\!x)}{2}\]

\[\int e^x\left(x\ln\!x-x-\frac{1}{x}\right)\ \mbox{d}x=e^x(x\ln\!x-x-\ln\!x)\]

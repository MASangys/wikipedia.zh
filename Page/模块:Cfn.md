function trim( str )

`   if str == nil then`
`       return nil;`
`   end`
`   return str:match( "^%s*(.-)%s*$");`

end

`   end`
`   `
`   if args.year ~= "" then`
`       result = result .. '（' .. args.year .. '年';`
`       if args.month ~= "" then`
`           result = result .. args.month .. '月';`

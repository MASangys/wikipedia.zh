local function core(name, value)

`   local ret`
`   `
`   ret = '`

<li>

<b>' .. (name or '') .. '：</b>' .. (value or '') .. '

</li>

'

`   return ret`

end

local function tasklist(args) -- table.insert(list, -1, args.othertext or '其它')

`   local ret = ''`
`   `
`   for _, v in ipairs(list) do`
`       if args[v[1]] then`
`           ret = ret .. core(v[2], args[v[1]]) `
`       end`
`   end`
`   `
`   return ret`

end

end

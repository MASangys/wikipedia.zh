local function getLabelName(input)

`   if input == nil then`
`       return ''`
`   end`
`   `
`   for key, value in ipairs(labels) do`
`       if input == value[1] then`
`           return value[2] .. '：'`
`       end`
`   end`
`   `
`   return input .. '：'`

end

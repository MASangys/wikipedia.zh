**藤田-斯托姆方程**(Fujita-Storm equation)是一个非线性偏微分方程：\[1\]

\(u_{t}=a*(u^{-2}*u_{x})_{x}\)

## 解析解

\[f1 := 1/\sqrt(2*C1*x-2*a*C1^2*t+C2)\]

\[f2 := -1/\sqrt(2*C1*x-2*a*C1^2*t+C2)\]

\[f3 := 1/\sqrt((1/2)*C1*(x+C2)^2/a+C3*\exp(C1*t))\]

\[f4 := -1/\sqrt((1/2)*C1*(x+C2)^2/a+C3*\exp(C1*t))\]

\[f5 := 1/(\sqrt(c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))*\sqrt(4-\exp(8*a*t)*(c*\exp(-a*t)+c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))))\]

\[f6 := 1/(\sqrt(c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))*\sqrt(4-\exp(8*a*t)*(c*\exp(-a*t)-c^2*\exp(-2*a*t)-2*\exp(-8*a*t)-2*x*\exp(-4*a*t))))\]

\[f7 := -1/(\sqrt(c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))*\sqrt(4-\exp(8*a*t)*(c*\exp(-a*t)+c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))))\]

\[f8 := -1/(\sqrt(c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))*\sqrt(4-\exp(8*a*t)*(c*\exp(-a*t)-c^2*\exp(-2*a*t)-2*\exp(-8*a*t)-2*x*\exp(-4*a*t))))\]

\[f9 := 1/(\sqrt(c^2*\exp(2*a*t)+2*\exp(8*a*t)+2*x*\exp(4*a*t))*\sqrt(\exp(8*a*t)*(c*\exp(-a*t)+c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))))\]

\[f10 := 1/(\sqrt(c^2*\exp(2*a*t)+2*\exp(8*a*t)+2*x*\exp(4*a*t))*\sqrt(\exp(8*a*t)*(c*\exp(-a*t)-c^2*\exp(-2*a*t)-2*\exp(-8*a*t)-2*x*\exp(-4*a*t))))\]

\[f11 := -1/(\sqrt(c^2*\exp(2*a*t)+2*\exp(8*a*t)+2*x*\exp(4*a*t))*\sqrt(\exp(8*a*t)*(c*\exp(-a*t)+c^2*\exp(-2*a*t)+2*\exp(-8*a*t)+2*x*\exp(-4*a*t))))\]

\[f12 := -1/(\sqrt(c^2*\exp(2*a*t)+2*\exp(8*a*t)+2*x*\exp(4*a*t))*\sqrt(\exp(8*a*t)*(c*\exp(-a*t)-c^2*\exp(-2*a*t)-2*\exp(-8*a*t)-2*x*\exp(-4*a*t))))\]
